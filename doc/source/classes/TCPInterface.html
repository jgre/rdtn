  def initialize(host = "localhost", port = 6666)
    @s = TCPServer.new(host,port)
    # register this socket
    @s.extend EventLoop::Watchable
    @s.will_block = false
    @s.on_readable { self.whenAccept }
    @s.monitor_event :readable
  end
